\begin{equation}
p(y|x,z)p(x|z) = \frac{p(x,y,z)}{p(x,z)} \frac {p(x,z)}{p(z)} = \frac{p(x,y,z)}{p(z)} = p(x,y|z)
\end{equation}


\begin{equation}
\frac{p(y|x,z)p(x|z)}{p(y|z)} = \frac{p(x,y,z)p(x,z)p(z)}{p(z)p(x,z)p(y,z)} = \frac{p(x,y,z)}{p(y,z)} = p(x|y,z)
\end{equation}

\end{equation}
