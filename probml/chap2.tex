ex 2.8
\begin{equation}
p(x|z) = \int_{y} p(x,y)dy = \int_{y}h(x,z)g(y,z)dy = h(x,z)\int_{y}g(y,z)dy = h(x,z)C_1
\end{equation}
similarly
\begin{equation}
p(y|z) = g(y,z)C_2
\end{equation}
it remains to show $C = C_1C_2 = 1$
\begin{equation}
1 = \int_{x,y}p(x,y|z) = C\int_{x}p(x|z)dx\int_{y}p(y|z)dy = C 
\end{equation}
